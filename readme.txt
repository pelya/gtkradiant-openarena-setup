If you are using Debian/Ubuntu, run following commands to install necessary packages:

sudo dpkg --add-architecture i386
sudo apt-get update
sudo apt-get install netcat scons make g++-multilib build-essential python libglib2.0-dev \
  libgtk2.0-dev libgtkglext1-dev libxml2-dev libpango1.0-dev libatk1.0-dev zlib1g-dev libpng12-dev
